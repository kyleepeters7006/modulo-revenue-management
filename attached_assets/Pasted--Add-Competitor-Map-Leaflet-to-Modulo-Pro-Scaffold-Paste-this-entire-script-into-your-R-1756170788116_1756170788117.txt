let chart, map, markers=[];

function bindSlider(id){ const el=document.getElementById(id); const out=el.nextElementSibling; el.addEventListener('input', ()=> out.value=el.value); out.value=el.value; }

async function init(){
  ["wOcc","wVac","wAttr","wSeas","wComp","wMkt"].forEach(bindSlider);
  uploadBtn.onclick = uploadRentRoll;
  saveAsm.onclick = saveAssumptions;
  saveWeights.onclick = saveWeightsFn;
  saveGuardrails.onclick = saveGuardrailsFn;
  publishBtn.onclick = publishRates;
  refreshMarket.onclick = fetchMarket;
  recompute.onclick = loadRecs;
  aiBtn.onclick = askAI;
  trainBtn.onclick = trainML;
  addComp.onclick = addCompetitor;

  await refreshStatus();
  await drawChart();
  await fetchMarket();
  await loadRecs();
  setupMap();      // NEW
  await refreshMap(); // NEW
  await loadComparison();
}

async function refreshStatus(){
  const r = await fetch('/api/status'); const d = await r.json();
  startRev.innerText = "Starting Revenue: $" + (d.starting_revenue?.toFixed(2) || 0);
  occ.innerText = "Occupancy: " + ((d.occupancy||0)*100).toFixed(1) + "%";
  // hydrate
  const a = d.assumptions || {};
  asmStart.value = a.start_period || "2025-09";
  asmMonths.value = a.months ?? 24;
  asmRev.value = a.revenue_monthly_growth_pct ?? 0.35;
  asmSPX.value = a.sp500_monthly_return_pct ?? 0.45;
  asmTarget.value = a.target_occupancy ?? 0.92;
  const w = d.weights || {};
  wOcc.value = w.occupancy_pressure ?? 25; wOcc.nextElementSibling.value = wOcc.value;
  wVac.value = w.days_vacant_decay ?? 20; wVac.nextElementSibling.value = wVac.value;
  wAttr.value = w.room_attributes ?? 25;  wAttr.nextElementSibling.value = wAttr.value;
  wSeas.value = w.seasonality ?? 10;      wSeas.nextElementSibling.value = wSeas.value;
  wComp.value = w.competitor_rates ?? 10; wComp.nextElementSibling.value = wComp.value;
  wMkt.value = w.stock_market ?? 10;      wMkt.nextElementSibling.value = wMkt.value;

  const g = await (await fetch('/api/guardrails')).json();
  guardrailsJson.value = JSON.stringify(g || {}, null, 2);
}

async function drawChart(){
  const r = await fetch('/api/series'); const d = await r.json();
  const ctx = document.getElementById('revChart').getContext('2d');
  if(chart) chart.destroy();
  chart = new Chart(ctx,{
    type:'line',
    data:{ labels:d.labels, datasets:[
      {label:'Revenue', data:d.revenue},
      {label:'S&P 500 (Indexed)', data:d.sp500}
    ]},
    options:{ responsive:true, maintainAspectRatio:false,
      scales:{ y:{ ticks:{ callback:v=>'$'+Number(v).toLocaleString() }}}}
  });
}

async function uploadRentRoll(){
  const f = rentRollFile.files[0]; if(!f){ alert('Choose a CSV first.'); return; }
  const fd = new FormData(); fd.append('file', f);
  const r = await fetch('/api/upload_rent_roll',{method:'POST', body:fd});
  const d = await r.json(); uploadMsg.innerText = d.error? ('Error: '+d.error) : 'Uploaded '+d.rows+' rows';
  await refreshStatus(); await drawChart(); await loadRecs(); await loadComparison();
}

async function saveAssumptions(){
  const payload = {
    start_period: asmStart.value,
    months: Number(asmMonths.value),
    revenue_monthly_growth_pct: Number(asmRev.value),
    sp500_monthly_return_pct: Number(asmSPX.value),
    target_occupancy: Number(asmTarget.value)
  };
  const r = await fetch('/api/assumptions',{method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify(payload)});
  const d = await r.json(); asmMsg.innerText = d.ok? 'Saved.' : 'Error';
  await drawChart(); await loadRecs();
}

async function saveWeightsFn(){
  const payload = {
    occupancy_pressure: Number(wOcc.value),
    days_vacant_decay: Number(wVac.value),
    room_attributes: Number(wAttr.value),
    seasonality: Number(wSeas.value),
    competitor_rates: Number(wComp.value),
    stock_market: Number(wMkt.value)
  };
  const r = await fetch('/api/weights',{method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify(payload)});
  const d = await r.json(); weightsMsg.innerText = d.ok? 'Weights saved.' : 'Error';
  await loadRecs();
}

async function saveGuardrailsFn(){
  try{
    const payload = JSON.parse(guardrailsJson.value || "{}");
    const r = await fetch('/api/guardrails',{method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify(payload)});
    const d = await r.json(); guardMsg.innerText = d.ok? 'Guardrails saved.' : 'Error';
    await loadRecs();
  }catch(e){ guardMsg.innerText = 'Invalid JSON'; }
}

async function fetchMarket(){
  const r = await fetch('/api/market'); const d = await r.json();
  mktBox.innerText = "Last monthly S&P return: " + (d.last_month_return_pct?.toFixed(2) ?? 0) + "%";
}

async function loadRecs(){
  const r = await fetch('/api/recommendations'); const d = await r.json();
  const tb = document.querySelector('#recTable tbody'); tb.innerHTML = '';
  (d.items||[]).forEach(u=>{
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td>${u.Unit_ID}</td>
      <td>${u.Room_Type}</td>
      <td>${u.Occupied_YN}</td>
      <td>${u.Days_Vacant ?? 0}</td>
      <td>${fmt(u.Fence_Price)}</td>
      <td>${fmt(u.Competitor_Benchmark_Rate)}</td>
      <td><b>${fmt(u.Recommended_Rent)}</b></td>
      <td>${u.ML_Suggested_Rent ? fmt(u.ML_Suggested_Rent) : ''}</td>
      <td>${u.Rationale || ''}</td>`;
    tb.appendChild(tr);
  });
}

async function loadComparison(){
  const r = await fetch('/api/compare'); const d = await r.json();
  const tb = document.querySelector('#cmpTable tbody'); tb.innerHTML = '';
  (d.rows||[]).forEach(row=>{
    const tr = document.createElement('tr');
    tr.innerHTML = `
      <td>${row.Room_Type||''}</td>
      <td>${fmt(row.Your_Current_Avg)}</td>
      <td>${fmt(row.Market_Avg)}</td>
      <td>${fmt(row.Competitor_Avg_Care)}</td>
      <td>${fmt(row.Net_vs_Market)}</td>
      <td>${fmt(row.Modulo_Recommended)}</td>`;
    tb.appendChild(tr);
  });
}

function fmt(v){ if(v===""||v===undefined||v===null) return ""; if(typeof v==='number') return '$'+Number(v).toFixed(2); return v }

async function publishRates(){
  const r = await fetch('/api/publish',{method:'POST'}); const d = await r.json();
  pubMsg.innerHTML = d.ok ? `Published: <a href="${d.csv}" target="_blank">download CSV</a>` : 'Error';
}

async function askAI(){
  aiOut.textContent = "Thinking...";
  const r = await fetch('/api/ai/suggest',{method:'POST'}); const d = await r.json();
  aiOut.textContent = d.ok ? d.text : ("Error: " + d.error);
}

async function trainML(){
  const f = histFile.files[0];
  let opts = {}; let url = '/api/ml/train';
  if(f){ const fd = new FormData(); fd.append('file', f); opts.method='POST'; opts.body=fd; }
  else { opts.method='POST'; }
  const r = await fetch(url, opts); const d = await r.json();
  mlMsg.textContent = d.ok ? `Trained (R^2=${Number(d.r2||0).toFixed(3)}, rows=${d.rows})` : (d.error || 'Train error');
  await loadRecs();
}

function setupMap(){
  // Center map on Louisville, KY
  map = L.map('map').setView([38.2527, -85.7585], 11);
  L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png',{
    attribution:'© OpenStreetMap'
  }).addTo(map);
}

async function refreshMap(){
  const r = await fetch('/api/competitors'); const d = await r.json();
  (markers||[]).forEach(m => map.removeLayer(m));
  markers = [];
  (d.items||[]).forEach(c=>{
    const m = L.marker([c.lat, c.lng]).addTo(map);
    const rates = c.rates ? Object.entries(c.rates).map(([rt,val])=>`${rt}: $${val}`).join('<br>') : '';
    m.bindPopup(`<b>${c.name}</b><br>${rates}<br>Avg Care: $${c.avg_care_rate ?? ''}`);
    markers.push(m);
  });
}

async function addCompetitor(){
  const name = cName.value.trim();
  const lat = Number(cLat.value), lng = Number(cLng.value);
  const rt = cRT.value.trim(), rate = Number(cRate.value), care = Number(cCare.value);
  if(!name || isNaN(lat) || isNaN(lng) || !rt || isNaN(rate)){ compMsg.innerText = 'Fill required fields.'; return; }
  const payload = { name, lat, lng, rates: {[rt]: rate} };
  if(!Number.isNaN(care) && care>0) payload.avg_care_rate = care;
  const r = await fetch('/api/competitors',{method:'POST', headers:{'Content-Type':'application/json'}, body:JSON.stringify(payload)});
  const d = await r.json(); compMsg.innerText = d.ok? 'Saved.' : 'Error';
  await refreshMap(); await loadComparison();
}

init();
